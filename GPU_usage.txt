import torch
import time
from IPython.display import display, clear_output
import matplotlib.pyplot as plt
import matplotlib.animation as animation
from collections import deque

# ── Config ─────────────────────────────────────────────────────────────────────
MAX_POINTS    = 100      # how many data points to show on x-axis
UPDATE_EVERY  = 0.5      # seconds between updates
TOTAL_VRAM_GB = torch.cuda.get_device_properties(0).total_memory / 1024**3

# ── Data buffers ───────────────────────────────────────────────────────────────
times      = deque(maxlen=MAX_POINTS)
used_gb    = deque(maxlen=MAX_POINTS)
reserved_gb= deque(maxlen=MAX_POINTS)
start_time = time.time()

# ── Plot setup ─────────────────────────────────────────────────────────────────
fig, ax = plt.subplots(figsize=(10, 4))
fig.patch.set_facecolor("#1e1e1e")
ax.set_facecolor("#1e1e1e")
ax.set_title("Live GPU Memory Usage", color="white", fontsize=13)
ax.set_xlabel("Time (s)",  color="white")
ax.set_ylabel("VRAM (GB)", color="white")
ax.tick_params(colors="white")
ax.set_ylim(0, TOTAL_VRAM_GB * 1.05)
for spine in ax.spines.values():
    spine.set_edgecolor("#444")

line_used,     = ax.plot([], [], color="#00d4ff", linewidth=2, label="Used")
line_reserved, = ax.plot([], [], color="#ff6b6b", linewidth=1.5,
                          linestyle="--", label="Reserved")
ax.axhline(y=TOTAL_VRAM_GB, color="#888", linestyle=":", linewidth=1,
           label=f"Total ({TOTAL_VRAM_GB:.1f} GB)")
ax.legend(facecolor="#2e2e2e", labelcolor="white", loc="upper left")

txt = ax.text(0.98, 0.95, "", transform=ax.transAxes,
              ha="right", va="top", color="white",
              fontsize=10, fontfamily="monospace",
              bbox=dict(facecolor="#2e2e2e", edgecolor="#555", boxstyle="round"))

plt.tight_layout()

# ── Update function ────────────────────────────────────────────────────────────
try:
    while True:
        elapsed   = time.time() - start_time
        used      = torch.cuda.memory_allocated(0)  / 1024**3
        reserved  = torch.cuda.memory_reserved(0)   / 1024**3

        times.append(elapsed)
        used_gb.append(used)
        reserved_gb.append(reserved)

        t_list = list(times)
        line_used.set_data(t_list, list(used_gb))
        line_reserved.set_data(t_list, list(reserved_gb))

        ax.set_xlim(max(0, elapsed - MAX_POINTS * UPDATE_EVERY), elapsed + 1)

        txt.set_text(
            f"Used     : {used:.2f} GB\n"
            f"Reserved : {reserved:.2f} GB\n"
            f"Free     : {TOTAL_VRAM_GB - reserved:.2f} GB\n"
            f"Total    : {TOTAL_VRAM_GB:.1f} GB"
        )

        clear_output(wait=True)
        display(fig)
        time.sleep(UPDATE_EVERY)

except KeyboardInterrupt:
    print("Monitor stopped.")
    plt.close()
