import torch
from transformers import AutoModelForCausalLM, AutoTokenizer

# Path to your locally saved Qwen model — must match QWEN_LOCAL_PATH above
QWEN_LOCAL_PATH = "./qwen_model"

_tokenizer = None
_model     = None

def _load_qwen():
    """Load Qwen once and cache it in module-level variables."""
    global _tokenizer, _model
    if _model is None:
        print("Loading Qwen from disk...")
        _tokenizer = AutoTokenizer.from_pretrained(QWEN_LOCAL_PATH)
        _model     = AutoModelForCausalLM.from_pretrained(
            QWEN_LOCAL_PATH,
            torch_dtype = torch.float16,
            device_map  = "auto",
        )
        _model.eval()
        print("Qwen loaded.")
    return _tokenizer, _model


def run_client(
    prompts: list[str],
    vllm_model: str,         # ignored — kept for API compatibility
    vllm_max_tokens: int,
) -> list[str]:
    """
    Drop-in replacement for the vLLM client.
    Runs Qwen2.5-7B-Instruct locally using HuggingFace Transformers.
    Works on Windows — no vLLM required.
    """
    tokenizer, model = _load_qwen()
    outputs = []

    for prompt in prompts:
        messages = [{"role": "user", "content": prompt}]

        # Apply Qwen chat template
        text = tokenizer.apply_chat_template(
            messages,
            tokenize            = False,
            add_generation_prompt = True,
        )

        inputs = tokenizer(text, return_tensors="pt").to(model.device)

        with torch.no_grad():
            generated = model.generate(
                **inputs,
                max_new_tokens = vllm_max_tokens,
                temperature    = 0.7,
                top_p          = 0.9,
                do_sample      = True,
                pad_token_id   = tokenizer.eos_token_id,
            )

        # Decode only the newly generated tokens
        new_tokens = generated[0][inputs["input_ids"].shape[1]:]
        result     = tokenizer.decode(new_tokens, skip_special_tokens=True).strip()
        outputs.append(result)
    return outputs
