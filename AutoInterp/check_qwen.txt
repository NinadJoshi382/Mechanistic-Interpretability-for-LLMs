import torch
from transformers import AutoModelForCausalLM, AutoTokenizer

# ── Load Qwen from local path ──────────────────────────────────────────────────
QWEN_LOCAL_PATH = "/path/to/your/qwen_model"   # <-- change this

tokenizer = AutoTokenizer.from_pretrained(QWEN_LOCAL_PATH)
model     = AutoModelForCausalLM.from_pretrained(
   QWEN_LOCAL_PATH,
   torch_dtype = torch.float16,
   device_map  = "auto",
)
model.eval()
print(f"Qwen loaded on: {next(model.parameters()).device}")

# ── Run a prompt ───────────────────────────────────────────────────────────────
prompt   = "Explain what a neural network is in two sentences."
messages = [{"role": "user", "content": prompt}]

text = tokenizer.apply_chat_template(
   messages,
   tokenize              = False,
   add_generation_prompt = True,
)

inputs = tokenizer(text, return_tensors="pt").to(model.device)

with torch.no_grad():
   generated = model.generate(
       **inputs,
       max_new_tokens = 200,
       temperature    = 0.7,
       top_p          = 0.9,
       do_sample      = True,
       pad_token_id   = tokenizer.eos_token_id,
   )

# Decode only the new tokens — not the prompt
new_tokens = generated[0][inputs["input_ids"].shape[1]:]
response   = tokenizer.decode(new_tokens, skip_special_tokens=True).strip()

print(f"\nPrompt   : {prompt}")
print(f"Response : {response}")
