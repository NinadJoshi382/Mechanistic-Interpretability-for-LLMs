    @classmethod
    def from_hooked_transformer_and_transcoders(
        cls,
        hooked_transformer,
        transcoders,
    ) -> "TransformerLensReplacementModel":
        """
        Create a TransformerLensReplacementModel from an already-loaded
        HookedTransformer — bypasses HuggingFace download entirely.
        Used for custom models like NanoGPT that aren't on HuggingFace.
        """
        instance = cls.__new__(cls)
        super(TransformerLensReplacementModel, instance).__init__(
            hooked_transformer.cfg
        )
        instance.load_state_dict(hooked_transformer.state_dict(), strict=False)
        instance._configure_replacement_model(transcoders)
        return instance
