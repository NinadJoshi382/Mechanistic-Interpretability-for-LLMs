# Cell 1 — Windows Application Event Log errors (last 2 hours)
import subprocess

result = subprocess.run(
    [
        "powershell", "-Command",
        """
        Get-WinEvent -LogName Application -MaxEvents 50 |
        Where-Object { $_.LevelDisplayName -eq 'Error' -or $_.LevelDisplayName -eq 'Critical' } |
        Where-Object { $_.TimeCreated -gt (Get-Date).AddHours(-2) } |
        Select-Object TimeCreated, LevelDisplayName, ProviderName, Message |
        Format-List
        """
    ],
    capture_output=True, text=True
)
print(result.stdout[:5000])
print(result.stderr[:1000])




# Cell 2 — Jupyter kernel log
import subprocess

result = subprocess.run(
    ["powershell", "-Command",
     "Get-ChildItem $env:APPDATA\\jupyter\\runtime\\ | Sort-Object LastWriteTime -Descending | Select-Object -First 5 | Format-List"],
    capture_output=True, text=True
)
print("Jupyter runtime files:")
print(result.stdout)

result2 = subprocess.run(
    ["powershell", "-Command",
     "Get-ChildItem $env:APPDATA\\jupyter\\runtime\\*.log | Sort-Object LastWriteTime -Descending | Select-Object -First 1 | Get-Content | Select-Object -Last 50"],
    capture_output=True, text=True
)
print("\nLast 50 lines of most recent Jupyter log:")
print(result2.stdout)



# Cell 3 — CUDA and GPU errors in System log
import subprocess

result = subprocess.run(
    [
        "powershell", "-Command",
        """
        Get-WinEvent -LogName System -MaxEvents 100 |
        Where-Object { $_.Message -like '*CUDA*' -or
                       $_.Message -like '*NVIDIA*' -or
                       $_.Message -like '*GPU*' -or
                       $_.Message -like '*display*' } |
        Where-Object { $_.TimeCreated -gt (Get-Date).AddHours(-2) } |
        Select-Object TimeCreated, LevelDisplayName, Message |
        Format-List
        """
    ],
    capture_output=True, text=True
)
print(result.stdout[:5000])



# Cell 4 — Python crash dumps
import subprocess

result = subprocess.run(
    ["powershell", "-Command",
     """
     Get-ChildItem $env:LOCALAPPDATA\\CrashDumps\\ -ErrorAction SilentlyContinue |
     Where-Object { $_.Name -like '*python*' } |
     Sort-Object LastWriteTime -Descending |
     Select-Object -First 5 |
     Format-List
     """],
    capture_output=True, text=True
)
print("Python crash dumps:")
print(result.stdout if result.stdout.strip() else "None found")
