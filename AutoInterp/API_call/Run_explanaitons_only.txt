        logger.info(f"[Job {job_id}] Done.")

    def run_explanations_only(
        self,
        *,
        job_id: int,
        save_dir: Optional[Path] = None,
    ) -> None:
        """
        Generate LLM explanations for already-saved parquet features.
        Call this after run(..., generate_explanations=False) has completed.
        No model forward passes — only reads saved parquet and calls the LLM.
        """
        if save_dir is None:
            if self.cfg.latent_cache_path is None:
                raise ValueError(
                    "Either pass save_dir or set cfg.latent_cache_path."
                )
            save_dir = Path(self.cfg.latent_cache_path)

        feature_dicts_by_layer = load_features_parquet(save_dir)
        logger.info(f"[Job {job_id}] Generating LLM explanations on saved features...")
        self._generate_and_add_explanations(
            feature_dicts_by_layer=feature_dicts_by_layer,
            job_id=job_id,
            save_dir=save_dir,
        )
        logger.info(f"[Job {job_id}] Explanations done.")
