    def run_explanations_only(
        self,
        *,
        job_id: int,
        save_dir: Optional[Path] = None,
    ) -> None:
        if save_dir is None:
            if self.cfg.latent_cache_path is None:
                raise ValueError(
                    "Either pass save_dir or set cfg.latent_cache_path."
                )
            save_dir = Path(self.cfg.latent_cache_path)

        n_layers = self.clt.N_layers
        feature_dicts_by_layer = []

        for layer in range(n_layers):
            rows = load_features_parquet(save_dir, layer=layer)
            layer_dict = {str(r["feature_index"]): r for r in rows}
            feature_dicts_by_layer.append(layer_dict)

        logger.info(f"[Job {job_id}] Loaded features for {n_layers} layers.")
        self._generate_and_add_explanations(
            feature_dicts_by_layer=feature_dicts_by_layer,
            job_id=job_id,
            save_dir=save_dir,
        )
        logger.info(f"[Job {job_id}] Explanations done.")
