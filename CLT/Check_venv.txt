# ── PyTorch CUDA Fix ──────────────────────────────────────────────────────────
# CLT-Forge requires torchvision >=0.22.0,<0.23.0  (from its pyproject.toml)
# torch 2.7.1 + torchvision 0.22.1 is the matching pair for CUDA 12.6
# Your driver supports CUDA 12.6, so we use the cu126 wheel index.

import importlib, sys, subprocess

TORCH_TARGET      = "2.7.1"
TORCHVISION_TARGET = "0.22.1"
CUDA_TAG          = "cu126"
TORCH_INDEX_URL   = f"https://download.pytorch.org/whl/{CUDA_TAG}"

GREEN  = "\033[92m"
YELLOW = "\033[93m"
RED    = "\033[91m"
RESET  = "\033[0m"
BOLD   = "\033[1m"

def get_installed_version(pip_name):
    try:
        from importlib.metadata import version
        return version(pip_name)
    except Exception:
        return None

def needs_reinstall(pkg, target_ver):
    """Returns True if pkg is missing OR is a CPU-only build OR version doesn't match."""
    inst = get_installed_version(pkg)
    if inst is None:
        return True, f"not installed"
    # Check for CPU-only build by inspecting the local version tag (+cpu)
    if "+cpu" in inst:
        return True, f"installed {inst} — CPU-only build, needs CUDA build"
    if inst != target_ver:
        return True, f"installed {inst}, need {target_ver}"
    # Version matches — but still verify CUDA is actually accessible
    if pkg == "torch":
        try:
            import torch
            if torch.version.cuda is None:
                return True, f"installed {inst} but torch.version.cuda is None (CPU-only build)"
        except ImportError:
            return True, "could not import torch"
    return False, f"installed {inst} — OK"

print(f"{BOLD}Checking PyTorch CUDA build for CUDA 12.6 / CLT-Forge compatibility...{RESET}\n")

reinstall_torch      = False
reinstall_torchvision = False

for pkg, target, label in [
    ("torch",       TORCH_TARGET,       "torch"),
    ("torchvision", TORCHVISION_TARGET, "torchvision"),
]:
    needs, reason = needs_reinstall(pkg, target)
    status_color  = RED if needs else GREEN
    status_text   = "REINSTALL NEEDED" if needs else "OK"
    print(f"  {label:<16} {status_color}{status_text:<18}{RESET} ({reason})")
    if pkg == "torch":
        reinstall_torch = needs
    else:
        reinstall_torchvision = needs

print()

if reinstall_torch or reinstall_torchvision:
    print(f"{YELLOW}Installing CUDA-enabled torch=={TORCH_TARGET} + "
          f"torchvision=={TORCHVISION_TARGET} for {CUDA_TAG}...{RESET}\n")

    # Uninstall CPU builds first to avoid version conflicts
    print("  Step 1/2 — removing existing CPU builds...")
    subprocess.call(
        [sys.executable, "-m", "pip", "uninstall", "-y", "torch", "torchvision", "torchaudio"],
        stdout=subprocess.DEVNULL, stderr=subprocess.DEVNULL,
    )

    # Install the correct CUDA-enabled wheels
    print("  Step 2/2 — installing CUDA wheels (this may take a few minutes)...")
    subprocess.check_call([
        sys.executable, "-m", "pip", "install", "-q",
        f"torch=={TORCH_TARGET}",
        f"torchvision=={TORCHVISION_TARGET}",
        "--index-url", TORCH_INDEX_URL,
    ])

    print(f"\n{GREEN}Installation complete. Verifying...{RESET}\n")

    # Re-import torch in a subprocess to get a clean check (avoids cached import)
    result = subprocess.run(
        [sys.executable, "-c",
         "import torch; print(torch.__version__); print(torch.version.cuda); "
         "print(torch.cuda.is_available()); "
         "print(torch.cuda.get_device_name(0) if torch.cuda.is_available() else 'no gpu')"],
        capture_output=True, text=True
    )
    lines = result.stdout.strip().splitlines()
    if len(lines) >= 3:
        print(f"  torch version  : {lines[0]}")
        print(f"  CUDA build     : {lines[1]}")
        print(f"  CUDA available : {lines[2]}")
        if len(lines) >= 4:
            print(f"  GPU            : {lines[3]}")

    print(f"\n{YELLOW}{BOLD}ACTION REQUIRED: Restart the kernel now, then re-run from Cell 1.{RESET}")
    print("(The currently running kernel still has the old CPU torch in memory.)")

else:
    import torch
    print(f"  torch CUDA build : {torch.version.cuda}")
    print(f"  CUDA available   : {torch.cuda.is_available()}")
    if torch.cuda.is_available():
        print(f"  GPU              : {torch.cuda.get_device_name(0)}")
    print(f"\n{GREEN}{BOLD}PyTorch is already correctly configured. No action needed.{RESET}")
