import torch, glob, os
from clt_forge.clt import CLT

# ── Find the latest valid checkpoint directory ─────────────────────────────────
ckpt_dirs = sorted([
    d for d in glob.glob(os.path.join(CLT_CHECKPOINT_DIR, "**"), recursive=True)
    if os.path.isdir(d)
    and os.path.exists(os.path.join(d, "clt_cfg.json"))
    and os.path.exists(os.path.join(d, "clt_weights.safetensors"))
])

if not ckpt_dirs:
    raise FileNotFoundError(
        f"No valid CLT checkpoint found under {CLT_CHECKPOINT_DIR}. "
        "Make sure Cell 10 completed successfully."
    )

BEST_CHECKPOINT = ckpt_dirs[-1]
print(f"Loading CLT from: {BEST_CHECKPOINT}")

clt_model = CLT.load_from_pretrained(BEST_CHECKPOINT, device=DEVICE)
clt_model.eval()
print("CLT loaded successfully.")

# ── Run a small eval batch through the frozen base model ───────────────────────
eval_tokens = torch.randint(0, VOCAB_SIZE, (4, CONTEXT_SIZE), device=DEVICE)

with torch.no_grad():
    _, cache = tl_model.run_with_cache(
        eval_tokens,
        names_filter=lambda n: "mlp" in n and ("hook_pre" in n or "hook_post" in n)
    )

# cache gives [batch, seq, d_in] per layer
# forward_eval expects [B, N_layers, d_in] where B = batch * seq
mlp_inputs  = [cache[f"blocks.{L}.mlp.hook_pre"]  for L in range(N_LAYER)]
mlp_outputs = [cache[f"blocks.{L}.mlp.hook_post"] for L in range(N_LAYER)]

# Stack layers → [batch, seq, N_layers, d_in] → flatten → [batch*seq, N_layers, d_in]
B, S = eval_tokens.shape
act_in  = torch.stack(mlp_inputs,  dim=2).view(B * S, N_LAYER, N_EMBD).to(DEVICE)
act_out = torch.stack(mlp_outputs, dim=2).view(B * S, N_LAYER, N_EMBD).to(DEVICE)

# ── Get CLT reconstruction ─────────────────────────────────────────────────────
with torch.no_grad():
    recon = clt_model.forward_eval(act_in)   # [B*S, N_layers, d_in]

# Split back into per-layer tensors for EV calculation
clt_recons = [recon[:, L, :] for L in range(N_LAYER)]
act_out_per_layer = [act_out[:, L, :] for L in range(N_LAYER)]

# ── Per-layer explained variance ───────────────────────────────────────────────
print("\nPer-layer explained variance (1.0 = perfect):")
total_ev = 0.0
for L in range(N_LAYER):
    target = act_out_per_layer[L].float()
    pred   = clt_recons[L].float()
    ss_res = ((target - pred) ** 2).sum().item()
    ss_tot = ((target - target.mean()) ** 2).sum().item()
    ev = 1.0 - ss_res / (ss_tot + 1e-8)
    total_ev += ev
    bar = '#' * int(max(0, ev) * 30)
    print(f"  Layer {L}: {ev:.4f}  |{bar:<30}|")
print(f"\n  Mean EV: {total_ev / N_LAYER:.4f}")
